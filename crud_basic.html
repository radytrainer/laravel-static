Route Function
Route::get('examples', [ExampleController::claas, 'index']); public function index() {
   return Example::all();
}
Route::get('examples/{id}', [ExampleController::claas, 'show']); public function show($id) {
   return Example::findOrFail();
}
Route::post('examples', [ExampleController::claas, 'store']); public function store(Request $request) {
   $test = new Example();
   $test->column1 = $request->input-name;
   $test->column2 = $request->input-name;
   $test->save();
   return response()->json(['sms' => 'Create successfully'], 201);
}
Route::update('examples/{id}', [ExampleController::claas, 'update']); public function update(Request $request, $id) {
   $test = Example::findOrFail($id);
   $test->column1 = $request->input-name;
   $test->column2 = $request->input-name;
   $test->save();
   return response()->json(['sms' => 'Update successfully'], 200);
}
Route::delete('examples/{id}', [ExampleController::claas, 'destroy']); public function destroy($id) {
   return Example::destroy($id);
}
